def duration(months):
    
    if months <= 10:
        return 'Less than one year'
    elif months >10 and months <=20:
        return 'Less than two years'
    elif months >20 and months <= 30:
        return 'Less than three years'
    elif months > 30 and months <= 40:
        return 'Less than four years'
    elif months > 40 and months <= 50:
        return 'Less than five years'
    elif months > 50 and months <= 60:
        return 'Less than six years'
    elif months > 60 and months <= 70:
        return 'Less then seven years'
